export const ContractItem = ({ name, link, address }) => (
  <li style={{ padding: 0, margin: 0 }}>

export const AssetItem = ({
  symbol,
  link,
  erc20,
  gateway,
  standard,
  decimals,
}) => (
  <li style={{ padding: 0, margin: 0 }}>
    <div style={{ border: "1px solid #eee", padding: 10, marginBottom: 2 }}>
      <b>
        {symbol}
        {decimals !== undefined ? <> ({decimals} decimals)</> : null}
      </b>
      <ul style={{ listStyleType: "none", padding: 0 }}>
        <li>
          {standard ? standard : <>ERC20</>}:{" "}
          <a href={`${link}${erc20}`}>{erc20}</a>
        </li>
        <li>
          Gateway: <a href={`${link}${gateway}`}>{gateway}</a>
        </li>
      </ul>
    </div>
  </li>
);
