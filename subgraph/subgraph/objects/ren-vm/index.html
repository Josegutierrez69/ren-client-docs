RenVM

The RenVM entity contains all the information about the current state of RenVM's on-chain information.

type RenVM {
id: ID!
activeBlock: BigInt!
activeTimestamp: BigInt!
previousActiveBlock: BigInt!
previousActiveTimestamp: BigInt!
numberOfDarknodes: BigInt!
numberOfDarknodesLastEpoch: BigInt!
numberOfDarknodesNextEpoch: BigInt!

minimumBond: BigInt!
minimumEpochInterval: BigInt!
currentEpoch: Epoch
previousEpoch: Epoch
deregistrationInterval: BigInt!
btcMintFee: Int!
btcBurnFee: Int!
zecMintFee: Int!
zecBurnFee: Int!
bchMintFee: Int!
bchBurnFee: Int!
mintFee(
skip: Int = 0
first: Int = 100
orderBy: ValueWithAsset_orderBy
orderDirection: OrderDirection
where: ValueWithAsset_filter
): [ValueWithAsset!]!
burnFee(
skip: Int = 0
first: Int = 100
orderBy: ValueWithAsset_orderBy
orderDirection: OrderDirection
where: ValueWithAsset_filter
): [ValueWithAsset!]!
currentCycle: BigInt!
previousCycle: BigInt!
currentCyclePayoutPercent: BigInt!

skip: Int = 0

totalTxCountBTC: BigInt!

totalLockedZEC: BigInt!

totalTxCountBCH: BigInt!
totalLockedBCH: BigInt!

}
